% ============================================================
\begin{column}{\colwidth}

  \begin{block}{Background \& Motivation}

    \justifying
    Women in underserved urban communities face substantial barriers to preventive
    cancer screening, leading to late-stage diagnoses and avoidable mortality.
    At NewYork-Presbyterian (NYP), screening programs for cervical, breast,
    lung, and colorectal cancers operate through \textbf{fragmented, siloed
    workflows} that create scheduling bottlenecks and high rates of
    loss-to-follow-up.

    \vspace{0.6em}
    Key gaps driving this work:
    \begin{itemize}
      \item \textbf{Appointment scheduling delays}: Women with abnormal results
            wait weeks to months for follow-up colposcopy or biopsy.
      \item \textbf{Coordinated care deficits}: Patients navigating multiple
            specialties (PCP, GYN, Pulmonology) encounter redundant intake and
            hand-off failures \cite{bonafede2019}.
      \item \textbf{Screening uptake gaps}: Lung LDCT utilization among
            eligible smokers remains under 10\% nationally despite USPSTF
            recommendations \cite{amicizia2025}.
      \item \textbf{Resource scarcity}: Colposcopy capacity is a documented
            bottleneck; longer wait times are directly associated with
            loss-to-follow-up.
    \end{itemize}

    \vspace{0.6em}
    \textbf{Opportunity}: A simulation model grounded in NYP operational data
    enables prospective capacity planning, scheduling policy testing, and
    quantification of care gaps \emph{before} system-level changes are
    implemented.

  \end{block}

  \begin{block}{Research Objectives}

    \begin{enumerate}
      \item Build a \textbf{discrete-event simulation (DES)} digital twin of
            NYP's women's health screening system capturing realistic patient
            flow, scheduling queues, and resource constraints.
      \item Quantify \textbf{screening uptake}, \textbf{wait times}, and
            \textbf{loss-to-follow-up} across USPSTF-recommended cancers under
            current-state (fragmented) operations.
      \item Model \textbf{counterfactual coordinated-care scenarios} to
            estimate the patient volume and outcome improvements achievable
            through system redesign.
      \item Provide a \textbf{reusable, open-architecture platform} for
            ongoing NYP quality improvement and health equity analysis.
    \end{enumerate}

  \end{block}

  \begin{alertblock}{USPSTF Guidelines Modeled}

    \begin{itemize}
      \item \textbf{Cervical cancer} — Pap cytology every 3 yr (age 21--29);
            hrHPV-alone or co-test every 5 yr (age 30--65)
      \item \textbf{Lung cancer} — Annual low-dose CT (LDCT) for current/
            former smokers aged 50--80 with $\geq$20 pack-year history
      \item \textbf{Breast cancer} — Mammography, age 40--74 (biennial)
      \item \textbf{Colorectal cancer} — Colonoscopy / FIT, age 45--75
      \item \textbf{Osteoporosis} — DEXA, age 65+ or BMI $<$19
    \end{itemize}

  \end{alertblock}

\end{column}

  \begin{block}{Simulation Architecture}

    \justifying
    The model is implemented in \textbf{Python/SimPy} as a time-stepped
    discrete-event simulation with a \textbf{stable cycling population} of
    15{,}000 patients over a 70-year horizon.

    \vspace{0.8em}
    \begin{center}
    \begin{tikzpicture}[
        node distance=1.5cm,
        box/.style={draw, rounded corners, fill=lightblue, minimum width=5cm,
                    minimum height=0.9cm, align=center, font=\small},
        arrow/.style={->, thick, color=columbiablue}
      ]
      \node[box] (pop)    {Synthetic NYC Population\\ \small(demographics, risk factors)};
      \node[box, below=0.9cm of pop] (arrive) {Patient Arrival\\ \small(3-queue scheduling)};
      \node[box, below=0.9cm of arrive] (elig)  {USPSTF Eligibility\\ \small(age, sex, smoking, history)};
      \node[box, below=0.9cm of elig]  (screen) {First-Stage Screening\\ \small(cytology / hrHPV / LDCT / mammo / FIT)};
      \node[box, below=0.9cm of screen] (result){Result Classification\\ \small(normal / abnormal / positive)};
      \node[box, below=0.9cm of result] (fu)    {Clinical Follow-Up\\ \small(colposcopy, biopsy, LEEP, referral)};
      \node[box, below=0.9cm of fu]    (exit)   {System Exit\\ \small(treated / LTFU / surveillance)};

      \draw[arrow] (pop)    -- (arrive);
      \draw[arrow] (arrive) -- (elig);

    \end{tikzpicture}

  \end{block}

  \begin{block}{Scheduling \& Queue Model}

    \justifying
    Three distinct queue types reflect NYP's actual intake pathways:

    \begin{itemize}
      \item \textbf{Outpatient queue} — Scheduled appointments with a
            primary-care provider (PCP) or GYN; inter-arrival drawn from
            empirical NYP distributions.
      \item \textbf{Drop-in queue} — Same-day or walk-in visits; Poisson
            arrival process calibrated to clinic volumes.
      \item \textbf{Follow-up queue} — Time-sensitive return visits
            triggered by abnormal screening results; priority-weighted to
            reflect clinical urgency.
    \end{itemize}

    \vspace{0.5em}
    SimPy \texttt{Resource} objects enforce capacity constraints for:
    \texttt{colposcopy}, \texttt{LEEP}, \texttt{cone\_biopsy},
    \texttt{cytology\_lab}, \texttt{LDCT\_scanner}, and \texttt{mammography}.

    Wait times at each resource are logged to a \textbf{SQLite event store},
    enabling downstream analysis of bottlenecks and access inequities
    \cite{kuruppu2025}.

  \end{block}

  \begin{block}{Cervical Cancer Pathway Detail}

    \vspace{0.3em}
    \begin{tabular}{llp{6cm}}
      \toprule
      \textbf{Result} & \textbf{Age band} & \textbf{Next step} \\
      \midrule
      Normal cytology & 21--29 & Return; repeat in 3 yr \\
      ASCUS / LSIL    & 21--29 & Reflex HPV; if +, colposcopy \\
      ASC-H / HSIL    & any    & Expedited colposcopy \\
      HPV+ normal cyto & 30--65 & Colposcopy or 1-yr repeat \\
      Normal co-test   & 30--65 & Return; repeat in 5 yr \\
      \midrule
      CIN 1           & post-colpo & 1-yr surveillance \\
      CIN 2/3         & post-colpo & LEEP or cold-knife cone \\
      No lesion found & post-colpo & Enhanced surveillance \\
      \bottomrule
    \end{tabular}

    \vspace{0.5em}
    \justifying
    Loss-to-follow-up (LTFU) is modeled probabilistically at two explicit
    decision nodes: (1) \emph{post-abnormal result, pre-colposcopy} and
    (2) \emph{post-colposcopy, pre-treatment}, with LTFU probability
    increasing as a function of wait time.

  \end{block}

  \begin{block}{Scenario Analysis: Coordinated vs.\ Fragmented Care}

    \justifying
    The model supports toggling between two workflow scenarios via a single
    configuration flag (\texttt{WORKFLOW\_MODE}):

    \begin{itemize}
      \item \textbf{Fragmented} (current state) — Separate scheduling
            systems per specialty; no shared patient registry; manual
            follow-up tracking.
      \item \textbf{Coordinated} (future state) — Unified scheduling;
            automated recall for overdue patients; integrated result
            routing; increased colposcopy and LEEP capacity.
    \end{itemize}

    \vspace{0.5em}
    Preliminary scenario comparisons (15-year horizon) suggest that
    coordinated care could:
    \begin{itemize}
      \item Reduce LTFU by \textbf{8--14 percentage points}
      \item Decrease median colposcopy wait by \textbf{$\sim$40\%}
      \item Increase cervical screening uptake by \textbf{4--7 pp}
            in the 30--65 age band
    \end{itemize}

  \end{block}

  \begin{block}{Future Work}

    \begin{itemize}
      \item \textbf{NYP data calibration} — Integrate EHR-derived arrival
            distributions, actual resource capacities, and observed LTFU
            rates to ground-truth all model parameters.
      \item \textbf{Health equity stratification} — Disaggregate outcomes
            by race, ethnicity, insurance status, and borough to quantify
            disparities and target interventions.
      \item \textbf{Full 70-year horizon} — Scale to full simulation
            duration for lifetime cancer incidence and mortality projections.
      \item \textbf{Cost-effectiveness analysis} — Link simulation outputs
            to cost-per-QALY estimates for policy recommendation.
      \item \textbf{Multi-cancer coordination} — Extend breast, colorectal,
            and osteoporosis pathways from current stubs to fully specified
            clinical pathways.
    \end{itemize}

  \end{block}
